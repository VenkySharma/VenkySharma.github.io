import requests
from bs4 import BeautifulSoup
import pandas as pd

url = "https://biharpolice.bih.nic.in/fir-data"
response = requests.get(url)
soup = BeautifulSoup(response.text, 'html.parser')

fir_counts = []
for row in soup.select('table tr')[1:]:
    cells = row.find_all('td')
    district = cells[0].text.strip()
    count = int(cells[1].text.strip())
    fir_counts.append({'District': district, 'FIR Count': count})

df = pd.DataFrame(fir_counts)
df.to_csv('bihar_fir_data.csv', index=False)
print("Data saved successfully!")
